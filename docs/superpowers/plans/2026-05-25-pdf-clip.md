# PDF 网摘 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add PDF clipping (网摘) support to RSS Pal across three entry points: browser-opened PDF via extension (A), local file:// PDF via extension (B), and remote PDF URL via the add-feed dialog (C). Extract text + images, fall back to OCR for scanned PDFs.

**Architecture:** New Go package `internal/pdfextract` shells out to poppler-utils (`pdftotext`, `pdfimages`, `pdftoppm`, `pdfinfo`) and Tesseract. Synchronous fast path returns immediately for digital PDFs; scanned PDFs go async via the worker. Images stored on filesystem under `${BACKUP_DIR}/article_images/<article_id>/` and served by a new JWT-protected endpoint with strong caching. Reuses existing `processing_state` column (already from migration 020) plus adds `processing_error` in migration 028.

**Tech Stack:** Go 1.24 (Gin, database/sql, lib/pq), Alpine Linux runtime, poppler-utils, tesseract-ocr (chi_sim + eng), React 18 + TypeScript + Vite, Chrome extension MV3 (manifest v3).

**Spec:** `docs/superpowers/specs/2026-05-25-pdf-clip-design.md` — read this first.

**Spec corrections (applied throughout this plan):**

1. The spec proposed adding `processing_status` — wrong, the column is named `processing_state` and already exists since migration 020. Reuse it. Only **add `processing_error TEXT`** in the new migration.
2. The spec said migration 028 — confirmed (master is at 026, weak-network-perf adds 027, we use 028 to avoid conflict).
3. The spec said `apt-get install` — wrong, Dockerfile is **Alpine**. Use `apk add`.
4. The worker query must include `parent_article_id IS NULL AND is_clip = true` to avoid colliding with link_set's existing use of `processing_state='processing'`.
5. The frontend has no separate "Add Feed Dialog" component — the add-feed UI lives inline in `FeedListPage.tsx`. The PDF entry will hook in there.

// Package pdfextract converts PDF bytes into structured text + images
// suitable for storage as a clipped article.
//
// It shells out to poppler-utils (pdftotext, pdfimages, pdftoppm, pdfinfo)
// and Tesseract via os/exec. Callers should treat extraction as IO-bound
// and avoid running many concurrent extractions on one machine.
package pdfextract

import "errors"

// Result is the full output of one extraction.
type Result struct {
	Title               string         // PDF metadata /Title → filename → URL
	Pages               []PageContent  // Per-page text + image refs
	Markdown            string         // Final assembled markdown
	TotalImagesOriginal int            // Before dedup
	TotalImagesKept     int            // After dedup + 100-cap
}

// PageContent is the per-page slice of a Result.
type PageContent struct {
	PageNum int         // 1-based
	Text    string      // Page text, no trailing whitespace
	Images  []ImageRef  // In page-order
}

// ImageRef describes one image extracted from the PDF. Bytes are loaded
// into memory; callers persist them via WriteImages.
type ImageRef struct {
	Idx    int    // 0-based unique per article
	Bytes  []byte // Raw image bytes (PNG or JPEG)
	Format string // "png" | "jpg"
	SHA256 string // Lowercase hex, for ETag + dedup
	Width  int    // Pixels; 0 if unknown
	Height int    // Pixels; 0 if unknown
}

// MaxImagesPerPDF caps how many images one article retains after dedup.
// Spec'd at 100; defined as a const so tests can lock to it.
const MaxImagesPerPDF = 100

// MinTextForDigital is the threshold below which ExtractFast considers a
// PDF "likely scanned" and the caller should fall back to ExtractWithOCR.
const MinTextForDigital = 200

// ErrNoText is returned by ExtractFast when fewer than MinTextForDigital
// runes of text are extracted across all pages. Callers should queue the
// PDF for async OCR via ExtractWithOCR.
var ErrNoText = errors.New("pdfextract: fewer than MinTextForDigital runes extracted; likely scanned")

// ExtractFast runs pdftotext + pdfimages + pdfinfo (no OCR). Returns
// ErrNoText (alongside a partial Result) when text is too sparse — caller
// should treat this as "queue for OCR" rather than a hard error.
func ExtractFast(pdfBytes []byte) (Result, error) {

// meta.go
package pdfextract

import (
	"bufio"
	"bytes"
	"fmt"
	"os/exec"
	"strings"
)

// extractTitle returns the PDF's /Title metadata, or "" if none. Returns
// a non-nil error when pdfinfo fails (corrupt PDF, missing binary).
func extractTitle(pdfBytes []byte) (string, error) {
	cmd := exec.Command("pdfinfo", "-")
	cmd.Stdin = bytes.NewReader(pdfBytes)
	var out, errBuf bytes.Buffer
	cmd.Stdout = &out
	cmd.Stderr = &errBuf
	if err := cmd.Run(); err != nil {
		return "", fmt.Errorf("pdfinfo: %w (stderr: %s)", err, strings.TrimSpace(errBuf.String()))
	}
	scanner := bufio.NewScanner(&out)
	for scanner.Scan() {
		line := scanner.Text()
		if strings.HasPrefix(line, "Title:") {
			return strings.TrimSpace(strings.TrimPrefix(line, "Title:")), nil
		}
	}
	if err := scanner.Err(); err != nil {
		return "", fmt.Errorf("scan pdfinfo output: %w", err)
	}
	return "", nil
}
```

// text.go
package pdfextract

import (
	"bytes"
	"fmt"
	"os/exec"
	"strings"
)

// extractTextPages runs `pdftotext -layout -nopgbrk` then splits the
// output on form feed (\f), which pdftotext emits between pages when
// -nopgbrk is omitted. We pass -nopgbrk OFF (the default) intentionally
// so we get the page boundaries. The flag name is misleading: passing
// -nopgbrk disables page breaks; omitting it keeps them.
func extractTextPages(pdfBytes []byte) ([]string, error) {
	cmd := exec.Command("pdftotext", "-layout", "-", "-")
	cmd.Stdin = bytes.NewReader(pdfBytes)
	var out, errBuf bytes.Buffer
	cmd.Stdout = &out
	cmd.Stderr = &errBuf
	if err := cmd.Run(); err != nil {
		return nil, fmt.Errorf("pdftotext: %w (stderr: %s)", err, strings.TrimSpace(errBuf.String()))
	}
	// Form feed (\f, 0x0C) is the page separator. The last page may or
	// may not have a trailing \f; strings.Split handles both cleanly.
	raw := strings.Split(out.String(), "\f")
	pages := make([]string, 0, len(raw))
	for _, p := range raw {
		trimmed := strings.TrimRight(p, " \t\n\r")
		// Keep empty pages — caller (markdown.go) decides whether to skip.
		pages = append(pages, trimmed)
	}
	// Drop a trailing empty page that's an artifact of a final \f.
	if len(pages) > 0 && pages[len(pages)-1] == "" {
		pages = pages[:len(pages)-1]
	}
	return pages, nil
}
```

package pdfextract

import (
	"bytes"
	"crypto/sha256"
	"encoding/hex"
	"fmt"
	"image"
	_ "image/jpeg"
	_ "image/png"

	"os/exec"
	"path/filepath"
	"sort"
	"strings"

// extractImages runs `pdfimages -all <pdf> <prefix>` on a temp file,
// reads each emitted image, deduplicates by SHA-256, sorts the survivors
// by page-number then within-page order, and caps the result at
// MaxImagesPerPDF. Returns the kept images plus the pre-dedup count so
// the caller can warn the user when truncation happened.
func extractImages(pdfBytes []byte) (kept []ImageRef, totalOriginal int, err error) {
	tmpDir, err := os.MkdirTemp("", "pdfextract-img-")
	if err != nil {
		return nil, 0, fmt.Errorf("mkdir temp: %w", err)
	}
	defer os.RemoveAll(tmpDir)

	pdfPath := filepath.Join(tmpDir, "in.pdf")
	if err := os.WriteFile(pdfPath, pdfBytes, 0o600); err != nil {
		return nil, 0, fmt.Errorf("write pdf: %w", err)
	}

	prefix := filepath.Join(tmpDir, "img")
	cmd := exec.Command("pdfimages", "-all", pdfPath, prefix)
	var errBuf bytes.Buffer
	cmd.Stderr = &errBuf
	if err := cmd.Run(); err != nil {
		return nil, 0, fmt.Errorf("pdfimages: %w (stderr: %s)", err, strings.TrimSpace(errBuf.String()))
	}

	entries, err := os.ReadDir(tmpDir)
	if err != nil {
		return nil, 0, fmt.Errorf("read tmpdir: %w", err)
	}

	// pdfimages names files like img-000.png, img-001.jpg, ... in page order.
	// Sort by filename ⇒ page order.
	files := make([]string, 0, len(entries))
	for _, e := range entries {
		name := e.Name()
		if name == "in.pdf" {
			continue
		}
		files = append(files, name)
	}
	sort.Strings(files)

	seenSHA := map[string]bool{}
	kept = make([]ImageRef, 0, MaxImagesPerPDF)
	idx := 0
	for _, name := range files {
		totalOriginal++
		if len(kept) >= MaxImagesPerPDF {
			continue // count but don't keep
		}
		b, err := os.ReadFile(filepath.Join(tmpDir, name))
		if err != nil {
			return nil, 0, fmt.Errorf("read image %s: %w", name, err)
		}
		sum := sha256.Sum256(b)
		shaHex := hex.EncodeToString(sum[:])
		if seenSHA[shaHex] {
			continue
		}
		seenSHA[shaHex] = true

		format := strings.TrimPrefix(filepath.Ext(name), ".")
		if format == "jpeg" {
			format = "jpg"
		}

		width, height := decodeDimensions(b)

		kept = append(kept, ImageRef{
			Idx:    idx,
			Bytes:  b,
			Format: format,
			SHA256: shaHex,
			Width:  width,
			Height: height,
		})
		idx++
	}
	return kept, totalOriginal, nil
}

// decodeDimensions returns width/height from the image header. Returns
// (0,0) if the image can't be decoded — non-fatal, we just lose the hint.
func decodeDimensions(b []byte) (int, int) {
	cfg, _, err := image.DecodeConfig(bytes.NewReader(b))
	if err != nil {
		return 0, 0
	}
	return cfg.Width, cfg.Height
}

```go
// markdown.go
package pdfextract

import (
	"fmt"
	"strings"
)

// assembleMarkdown stitches per-page text + image refs into the final
// article body. articleID is needed for image URLs. totalOriginal and
// totalKept drive the optional footer when the 100-image cap fired.
//
// Page-level invariants:
//   - A page with no text AND no images is skipped (no heading)
//   - A page with only text or only images still gets its heading
//   - Images render after the page's text, in slice order
func assembleMarkdown(pages []PageContent, articleID, totalImagesOriginal, totalImagesKept int) string {
	var b strings.Builder
	for _, p := range pages {
		hasText := strings.TrimSpace(p.Text) != ""
		hasImages := len(p.Images) > 0
		if !hasText && !hasImages {
			continue
		}
		fmt.Fprintf(&b, "## 第 %d 页\n\n", p.PageNum)
		if hasText {
			b.WriteString(p.Text)
			b.WriteString("\n\n")
		}
		for _, img := range p.Images {
			fmt.Fprintf(&b, "![](/api/articles/%d/images/%d.%s)\n", articleID, img.Idx, img.Format)
		}
		if hasImages {
			b.WriteString("\n")
		}
	}
	if totalImagesOriginal > totalImagesKept {
		dropped := totalImagesOriginal - totalImagesKept
		fmt.Fprintf(&b,
			"\n> 注：原 PDF 共 %d 张图（去重后 %d 张），超出 %d 张限制，已省略后 %d 张。\n",
			totalImagesOriginal, totalImagesKept, MaxImagesPerPDF, dropped)
	}
	return b.String()
}

// ocr.go
package pdfextract

import (
	"bytes"
	"fmt"
	"os"
	"os/exec"
	"path/filepath"
	"sort"
	"strings"
)

// ExtractWithOCR renders each page of the PDF to a 300 dpi PNG via
// pdftoppm, then OCRs each PNG via tesseract (chi_sim+eng). It also runs
// extractImages so we still get embedded raster images even for scanned
// PDFs. Per-page OCR failures get inlined into the page text as a marker
// line and the extraction still succeeds; only a pipeline-level failure
// (e.g. pdftoppm can't open the PDF) returns a non-nil error.
func ExtractWithOCR(pdfBytes []byte) (Result, error) {
	var r Result

	title, err := extractTitle(pdfBytes)
	if err != nil {
		return r, fmt.Errorf("title: %w", err)
	}
	r.Title = title

	tmpDir, err := os.MkdirTemp("", "pdfextract-ocr-")
	if err != nil {
		return r, fmt.Errorf("mkdir temp: %w", err)
	}
	defer os.RemoveAll(tmpDir)

	pdfPath := filepath.Join(tmpDir, "in.pdf")
	if err := os.WriteFile(pdfPath, pdfBytes, 0o600); err != nil {
		return r, fmt.Errorf("write pdf: %w", err)
	}

	pagePrefix := filepath.Join(tmpDir, "page")
	if out, err := exec.Command("pdftoppm", "-r", "300", "-png", pdfPath, pagePrefix).CombinedOutput(); err != nil {
		return r, fmt.Errorf("pdftoppm: %w (stderr: %s)", err, strings.TrimSpace(string(out)))
	}

	entries, err := os.ReadDir(tmpDir)
	if err != nil {
		return r, fmt.Errorf("read tmpdir: %w", err)
	}
	var pngs []string
	for _, e := range entries {
		if filepath.Ext(e.Name()) == ".png" {
			pngs = append(pngs, e.Name())
		}
	}
	sort.Strings(pngs)

	pages := make([]PageContent, 0, len(pngs))
	for i, name := range pngs {
		pageNum := i + 1
		imgPath := filepath.Join(tmpDir, name)
		var stdout, stderr bytes.Buffer
		cmd := exec.Command("tesseract", imgPath, "-", "-l", "chi_sim+eng")
		cmd.Stdout = &stdout
		cmd.Stderr = &stderr
		if err := cmd.Run(); err != nil {
			pages = append(pages, PageContent{
				PageNum: pageNum,
				Text: fmt.Sprintf("> [第 %d 页 OCR 失败：%s]",
					pageNum, strings.TrimSpace(stderr.String())),
			})
			continue
		}
		pages = append(pages, PageContent{
			PageNum: pageNum,

package pdfextract

import (
	"fmt"
	"os"
	"path/filepath"
	"strconv"
)

// imageDir returns the directory holding an article's PDF images.
func imageDir(base string, articleID int) string {
	return filepath.Join(base, "article_images", strconv.Itoa(articleID))
}

// ImagePath returns the absolute path for one image file.
func ImagePath(base string, articleID, idx int, format string) string {
	return filepath.Join(imageDir(base, articleID), strconv.Itoa(idx)+"."+format)
}

// WriteImages mkdir-p's the article's image dir and writes all images.
// Existing files at the same path are overwritten.
func WriteImages(base string, articleID int, imgs []ImageRef) error {
	dir := imageDir(base, articleID)
	if err := os.MkdirAll(dir, 0o755); err != nil {
		return fmt.Errorf("mkdir %s: %w", dir, err)
	}
	for _, im := range imgs {
		p := ImagePath(base, articleID, im.Idx, im.Format)
		if err := os.WriteFile(p, im.Bytes, 0o644); err != nil {
			return fmt.Errorf("write %s: %w", p, err)
		}
	}
	return nil
}

// RemoveImageDir deletes an article's entire image directory. Safe to
// call when the dir doesn't exist (returns nil).
func RemoveImageDir(base string, articleID int) error {
	dir := imageDir(base, articleID)
	if err := os.RemoveAll(dir); err != nil {
		return fmt.Errorf("rm %s: %w", dir, err)
	}
	return nil
}

// article_images.go
package api

import (
	"crypto/sha256"
	"encoding/hex"
	"net/http"
	"os"
	"path/filepath"
	"strconv"
	"strings"

	"github.com/gin-gonic/gin"
)

// AccessCheck verifies that the caller in c is allowed to read articleID.
// Returns (true, nil) when allowed, (false, nil) when denied, or
// (false, err) on infrastructure errors. Implemented as a func type so
// tests can inject simple stubs without spinning up the full auth stack.
type AccessCheck func(c *gin.Context, articleID int) (allowed bool, err error)

type ArticleImageHandler struct {
	baseDir string
	access  AccessCheck
}

func NewArticleImageHandler(baseDir string, access AccessCheck) *ArticleImageHandler {
	return &ArticleImageHandler{baseDir: baseDir, access: access}
}

// Serve responds to GET /api/articles/:id/images/:idx where :idx is
// e.g. "3.png" or "0.jpg". Returns the file with strong cache headers
// + ETag, supports If-None-Match for 304.
func (h *ArticleImageHandler) Serve(c *gin.Context) {
	idStr := c.Param("id")
	idxStr := c.Param("idx")
	articleID, err := strconv.Atoi(idStr)
	if err != nil {
		c.AbortWithStatus(http.StatusBadRequest)
		return
	}

	allowed, err := h.access(c, articleID)
	if err != nil {
		c.AbortWithStatus(http.StatusInternalServerError)
		return
	}
	if !allowed {
		c.AbortWithStatus(http.StatusForbidden)
		return
	}

	// idxStr looks like "3.png" — split into index + extension.
	dot := strings.LastIndex(idxStr, ".")
	if dot <= 0 {
		c.AbortWithStatus(http.StatusBadRequest)
		return
	}
	idx, err := strconv.Atoi(idxStr[:dot])
	if err != nil {
		c.AbortWithStatus(http.StatusBadRequest)
		return
	}
	ext := strings.ToLower(idxStr[dot+1:])
	contentType := "application/octet-stream"
	switch ext {
	case "png":
		contentType = "image/png"
	case "jpg", "jpeg":
		contentType = "image/jpeg"
	}

	path := filepath.Join(h.baseDir, "article_images", strconv.Itoa(articleID), strconv.Itoa(idx)+"."+ext)
	body, err := os.ReadFile(path)
	if err != nil {
		c.AbortWithStatus(http.StatusNotFound)
		return
	}

	sum := sha256.Sum256(body)
	etag := `"` + hex.EncodeToString(sum[:8]) + `"`
	if c.GetHeader("If-None-Match") == etag {
		c.Status(http.StatusNotModified)
		return
	}

	c.Header("Cache-Control", "public, max-age=31536000, immutable")
	c.Header("ETag", etag)
	c.Data(http.StatusOK, contentType, body)
}
```

type pdfCaptureResult struct {
	Status    string `json:"status"`     // "created" | "updated" | "processing"
	ArticleID int    `json:"article_id"`
	Message   string `json:"message"`
}

// processPDFCapture is the shared core for both capture-pdf and
// capture-pdf-url. It extracts the PDF (sync fast path), and either
// completes the article in one shot or stashes it as 'processing' for
// the worker to OCR.
func (h *BookmarkletHandler) processPDFCapture(
	user *model.User,
	rawURL, browserTitle string,
	pdfBytes []byte,
	imageBaseDir string,
) (pdfCaptureResult, error) {
	normalized := util.NormalizeURLKeepFragment(rawURL)

	// 1. Sync-attempt extraction.
	r, fastErr := pdfextract.ExtractFast(pdfBytes)
	queueForOCR := false
	if fastErr != nil {
		if !errorsIsNoText(fastErr) {
			return pdfCaptureResult{}, fmt.Errorf("extract: %w", fastErr)
		}
		queueForOCR = true
	}

	// 2. Pick a title. Priority: PDF metadata → browserTitle → filename → URL.
	title := strings.TrimSpace(r.Title)
	if title == "" {
		title = strings.TrimSpace(browserTitle)
	}
	if title == "" {
		title = filenameFromURL(rawURL)
	}
	if title == "" {
		title = normalized
	}

	// 3. Lookup/get the user's clip feed.
	feed, err := h.feedRepo.GetOrCreateClipFeed(user.ID)
	if err != nil {
		return pdfCaptureResult{}, fmt.Errorf("clip feed: %w", err)
	}

	// 4. Dedup: existing article for this user/URL?
	existing, err := h.articleRepo.FindByOwnerAndURL(user.ID, normalized)
	if err != nil {
		return pdfCaptureResult{}, fmt.Errorf("lookup: %w", err)
	}

	// 5a. Async branch — create stub, return processing.
	if queueForOCR {
		if existing != nil {
			// Reset existing back to processing so the worker re-OCRs it.
			if err := h.articleRepo.MarkPDFFailed(existing.ID, ""); err != nil {
				log.Printf("pdf: reset existing %d to failed (before re-queue): %v", existing.ID, err)
			}
			return pdfCaptureResult{Status: "processing", ArticleID: existing.ID, Message: "已重新加入 OCR 队列"}, nil
		}
		article := &model.Article{
			FeedID: feed.ID,
			Title:  title,
			URL:    normalized,
			IsClip: true,
		}
		if err := h.articleRepo.CreatePDFStub(article); err != nil {
			return pdfCaptureResult{}, fmt.Errorf("create stub: %w", err)
		}
		// Still extract images now (synchronous, cheap) so the figures
		// are available even before OCR completes.
		_ = pdfextract.WriteImages(imageBaseDir, article.ID, collectImages(r.Pages))
		return pdfCaptureResult{Status: "processing", ArticleID: article.ID, Message: "已加入 OCR 队列：" + title}, nil
	}

	// 5b. Sync branch — full content available immediately.
	var article *model.Article
	if existing != nil {
		article = existing
	} else {
		article = &model.Article{
			FeedID: feed.ID,
			Title:  title,
			URL:    normalized,
			IsClip: true,
		}
		if err := h.articleRepo.Create(article); err != nil {
			return pdfCaptureResult{}, fmt.Errorf("create: %w", err)
		}
	}

	// Write images first so the markdown URLs resolve.
	imgs := collectImages(r.Pages)
	if err := pdfextract.WriteImages(imageBaseDir, article.ID, imgs); err != nil {
		log.Printf("pdf: write images for article=%d failed: %v", article.ID, err)
		// Non-fatal: text content still useful.
	}

	pdfextract.BuildMarkdown(&r, article.ID)
	content := r.Markdown
	if len(content) > 50000 {
		content = content[:50000] + "..."
	}
	wc, rm := rss.ComputeMetrics(content)
	if err := h.articleRepo.UpdateContent(article.ID, content, wc, rm); err != nil {
		return pdfCaptureResult{}, fmt.Errorf("update content: %w", err)
	}
	if title != article.Title {
		_ = h.articleRepo.UpdateTitle(article.ID, title)
	}
	_ = h.articleRepo.UpdateSummary(article.ID, "", "") // trigger backfillSummaries

	status := "created"
	if existing != nil {
		status = "updated"
	}
	return pdfCaptureResult{Status: status, ArticleID: article.ID, Message: "已加入网摘：" + title}, nil
}

func collectImages(pages []pdfextract.PageContent) []pdfextract.ImageRef {
	var imgs []pdfextract.ImageRef
	for _, p := range pages {
		imgs = append(imgs, p.Images...)
	}
	return imgs
}

const captureMaxPDFBytes = 32 << 20

func (h *BookmarkletHandler) CapturePDF(c *gin.Context) {

		c.JSON(http.StatusUnauthorized, gin.H{"error": "无效的 bookmarklet token"})
		return
	}

	c.Request.Body = http.MaxBytesReader(c.Writer, c.Request.Body, captureMaxPDFBytes)
	if err := c.Request.ParseMultipartForm(captureMaxPDFBytes); err != nil {
		c.JSON(http.StatusRequestEntityTooLarge, gin.H{"error": "PDF 过大或表单解析失败"})
		return
	}

	rawURL := strings.TrimSpace(c.Request.FormValue("url"))
	if rawURL == "" {
		c.JSON(http.StatusBadRequest, gin.H{"error": "url 必填"})
		return
	}
	browserTitle := strings.TrimSpace(c.Request.FormValue("title"))

	fileHeader, err := c.FormFile("file")
	if err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": "file 必填"})
		return
	}
	if fileHeader.Size > captureMaxPDFBytes {
		c.JSON(http.StatusRequestEntityTooLarge, gin.H{"error": "PDF 过大"})
		return
	}
	f, err := fileHeader.Open()
	if err != nil {

		return
	}
	defer f.Close()
	pdfBytes, err := io.ReadAll(f)

		c.JSON(http.StatusInternalServerError, gin.H{"error": "读取 PDF 失败"})
		return

	} else if res.Status == "processing" {

// pdfFetchTimeout caps the upstream fetch. 30s is long enough for a slow
// CDN serving a 30 MB PDF on a flaky line, short enough that the user
// gets a clear error rather than a hung request.
const pdfFetchTimeout = 30 * time.Second

// CapturePDFURL is POST /api/bookmarklet/capture-pdf-url. Body:
// {"url": "..."}.
func (h *BookmarkletHandler) CapturePDFURL(c *gin.Context) {
	user, err := h.authenticate(c)
	if err != nil {
		c.JSON(http.StatusUnauthorized, gin.H{"error": "无效的 bookmarklet token"})
		return
	}

	var req struct {
		URL string `json:"url"`
	}
	if err := json.NewDecoder(c.Request.Body).Decode(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": "请求格式错误"})
		return
	}
	req.URL = strings.TrimSpace(req.URL)
	if req.URL == "" {
		c.JSON(http.StatusBadRequest, gin.H{"error": "url 必填"})
		return
	}

	client := &http.Client{Timeout: pdfFetchTimeout}
	httpReq, err := http.NewRequestWithContext(c.Request.Context(), "GET", req.URL, nil)
	if err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": "url 无效"})
		return
	}
	httpReq.Header.Set("User-Agent", "RSS-Pal-PDF-Fetch/1.0")
	resp, err := client.Do(httpReq)
	if err != nil {
		c.JSON(http.StatusBadGateway, gin.H{"error": "下载失败：" + err.Error()})
		return
	}
	defer resp.Body.Close()
	if resp.StatusCode != http.StatusOK {
		c.JSON(http.StatusBadGateway, gin.H{"error": fmt.Sprintf("下载失败：HTTP %d", resp.StatusCode)})
		return
	}
	ct := resp.Header.Get("Content-Type")
	if !strings.HasPrefix(ct, "application/pdf") && !strings.HasSuffix(strings.ToLower(req.URL), ".pdf") {
		c.JSON(http.StatusBadRequest, gin.H{"error": "目标不是 PDF：Content-Type=" + ct})
		return
	}
	pdfBytes, err := io.ReadAll(io.LimitReader(resp.Body, captureMaxPDFBytes+1))
	if err != nil {
		c.JSON(http.StatusBadGateway, gin.H{"error": "读取响应失败"})
		return
	}
	if int64(len(pdfBytes)) > captureMaxPDFBytes {
		c.JSON(http.StatusRequestEntityTooLarge, gin.H{"error": "PDF 过大"})
		return
	}

	res, err := h.processPDFCapture(user, req.URL, "", pdfBytes, h.imageBaseDir)
	if err != nil {
		log.Printf("CapturePDFURL user=%d url=%s: %v", user.ID, req.URL, err)
		c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
		return
	}
	status := http.StatusCreated
	if res.Status == "updated" {
		status = http.StatusOK
	} else if res.Status == "processing" {
		status = http.StatusAccepted
	}
	if h.backup != nil {
		h.backup.TriggerAsync()
	}
	c.JSON(status, res)
}

// pdf_ocr.go
package main

import (
	"context"
	"log"
	"time"

	"github.com/bytedance/rss-pal/internal/config"
	"github.com/bytedance/rss-pal/internal/pdfextract"
	"github.com/bytedance/rss-pal/internal/repository"
	"github.com/bytedance/rss-pal/internal/rss"
)

const maxPDFOCRPerCycle = 5

// processPDFOCR scans for PDF clip articles in processing state and runs
// the slow OCR path. Concurrency is intentionally 1 — OCR is CPU-bound
// and we don't want to thrash the box. The user-facing 'processing'
// badge gives clear feedback that something will arrive shortly.
func processPDFOCR(
	ctx context.Context,
	articleRepo *repository.ArticleRepository,
	cfg config.Config,
) {
	pending, err := articleRepo.GetPDFOCRPending(maxPDFOCRPerCycle)
	if err != nil {
		log.Printf("pdf_ocr: list pending: %v", err)
		return
	}
	if len(pending) == 0 {
		return
	}
	log.Printf("pdf_ocr: %d pending", len(pending))

	for i := range pending {
		select {
		case <-ctx.Done():
			return
		default:
		}
		a := &pending[i]
		runOnePDFOCR(a, articleRepo, cfg)
	}
}

func runOnePDFOCR(
	a *model.Article,
	articleRepo *repository.ArticleRepository,
	cfg config.Config,
) {
	// We need the original PDF bytes — but we don't store them. Strategy:
	// for HTTPS URLs we re-fetch; for file:// we cannot (worker has no
	// access to the user's local disk), so mark failed with a clear msg.
	//
	// This is a known tradeoff (see spec §Known limitations). The local-
	// file scanned-PDF workflow forces the user to redo the upload.
	if !looksLikeRemoteURL(a.URL) {
		_ = articleRepo.MarkPDFFailed(a.ID, "本地 PDF 异步 OCR 不可用：worker 无法读取 file:// 路径。请重新通过扩展上传，或改用浏览器打开 https 来源后再网摘。")
		return
	}

	pdfBytes, err := refetchPDF(a.URL)
	if err != nil {
		_ = articleRepo.MarkPDFFailed(a.ID, "重新下载 PDF 失败："+err.Error())
		return
	}
	r, err := pdfextract.ExtractWithOCR(pdfBytes)
	if err != nil {
		_ = articleRepo.MarkPDFFailed(a.ID, "OCR 失败："+err.Error())
		return
	}
	if err := pdfextract.WriteImages(cfg.Backup.Dir, a.ID, collectImagesFromPages(r.Pages)); err != nil {
		log.Printf("pdf_ocr: write images for %d: %v", a.ID, err)
	}
	pdfextract.BuildMarkdown(&r, a.ID)
	content := r.Markdown
	if len(content) > 50000 {
		content = content[:50000] + "..."
	}
	wc, rm := rss.ComputeMetrics(content)
	if err := articleRepo.UpdatePDFContent(a.ID, content, wc, rm); err != nil {
		log.Printf("pdf_ocr: UpdatePDFContent %d: %v", a.ID, err)
		_ = articleRepo.MarkPDFFailed(a.ID, "写入失败："+err.Error())
		return
	}
	if r.Title != "" && r.Title != a.Title {
		_ = articleRepo.UpdateTitle(a.ID, r.Title)
	}
	log.Printf("pdf_ocr: %d ready (pages=%d images=%d)", a.ID, len(r.Pages), r.TotalImagesKept)
}

func looksLikeRemoteURL(u string) bool {
	return len(u) > 8 && (u[:7] == "http://" || u[:8] == "https://")
}

func refetchPDF(url string) ([]byte, error) {
	client := &http.Client{Timeout: 30 * time.Second}
	resp, err := client.Get(url)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	if resp.StatusCode != 200 {
		return nil, fmt.Errorf("HTTP %d", resp.StatusCode)
	}
	return io.ReadAll(io.LimitReader(resp.Body, 32<<20))
}

func collectImagesFromPages(pages []pdfextract.PageContent) []pdfextract.ImageRef {
	var out []pdfextract.ImageRef
	for _, p := range pages {
		out = append(out, p.Images...)
	}
	return out
}
